CREATE TABLE public.hall (
  id INTEGER DEFAULT nextval('cafe_index_id_seq'::regclass) NOT NULL, 
  "LastUpdated" TIME WITHOUT TIME ZONE DEFAULT '12:00:00'::time without time zone NOT NULL, 
  code VARCHAR[], 
  "FullName" VARCHAR[], 
  "Location" VARCHAR[], 
  "Tel" CHAR(11)[], 
  "Email" VARCHAR[], 
  CONSTRAINT cafe_pkey PRIMARY KEY(id)
) WITHOUT OIDS;

CREATE UNIQUE INDEX cafe_code_key ON public.hall
  USING btree (code COLLATE pg_catalog."default");

CREATE TABLE public.hours (
  id INTEGER DEFAULT nextval('"Hours_id_seq"'::regclass) NOT NULL, 
  cafe_code VARCHAR[], 
  bld CHAR(1), 
  start_time TIME(0) WITHOUT TIME ZONE, 
  end_time TIME(0) WITHOUT TIME ZONE, 
  day CHAR(2)[], 
  CONSTRAINT "Hours_pkey" PRIMARY KEY(id), 
  CONSTRAINT "Hours_fk" FOREIGN KEY (cafe_code)
    REFERENCES public.hall(code)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION
    NOT DEFERRABLE
) WITHOUT OIDS;

CREATE TABLE public.meal (
  id INTEGER DEFAULT nextval('food_id_seq'::regclass) NOT NULL, 
  category VARCHAR[] NOT NULL, 
  "Name" VARCHAR[], 
  bld CHAR(1), 
  price REAL, 
  CONSTRAINT food_pkey PRIMARY KEY(id)
) WITHOUT OIDS;

COMMENT ON COLUMN public.meal.bld
IS '''b'' for breakfast
''l'' for lunch
''d'' for dinner
''o'' for other';
